# ХедХантер поиск

---Git и Github---
--терминал--
git init
    # стартовая инициализация Гит

git status
    # проверить статус файлов работающих с Гит

git add $НАЗВАНИЕ_ФАЙЛА.расширение$
    # добавить фай для отслеживания Гитом
        (для игнорирования ненужных или папок файлов создать файл - ".gitignore" и вписать в него например строчку - ".idea/",
        для добавления всех файлов в терминал вписать "git add .")

git commit -m "Краткое описание первых изменений в проекте"
    # закомитить первые изменения

СОЗДАЕМ НОВЫЙ РЕПОЗИТОРИЙ(проект) НА САЙТЕ GitHub и сохраняем нужные ссылки

git remote add origin $ссылка на проект взята с GitHub$
    # связываем GitHub с вашей рабочей средой

git remote

git push -u origin master
    # отправка изменений на GitHub

!!!ГОТОВО!!!

git clone $ссылка на GitHub проект$
    # послучить(скачать) себе проект с GitHub



git push origin master
    # отправить изменения на GitHub
